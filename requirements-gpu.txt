# Extra deps for the optional local GPU vision-LLM OCR backend.
# torch + CUDA come from the pytorch base image in Dockerfile.gpu.
# transformers pinned to 4.49.0, not just "< some version that needs
# gptqmodel": Qwen2_5_VLForConditionalGeneration was added in exactly
# 4.49.0, and 4.50.0+ introduced a well-documented regression breaking AWQ
# dispatch for Qwen2.5-VL specifically (surfaces as Triton CompilationErrors,
# CUDA asserts, garbled output, or dtype-mismatch RuntimeErrors depending on
# the exact 4.50+ version — see github.com/QwenLM/Qwen3-VL/issues/1033 and
# huggingface.co/Qwen/Qwen2.5-VL-7B-Instruct-AWQ/discussions/7). 4.49.0 is
# the narrow window that supports this model AND predates that regression
# AND predates newer releases' gptqmodel/CUDA-13 requirement.
transformers==4.49.0
accelerate>=1.0
qwen-vl-utils>=0.0.8
# Pinned exactly, not >=: older autoawq releases (0.2.6-0.2.8) each carry
# their own narrower, mutually-incompatible transformers/torch pins, which
# sends pip's resolver backtracking through all of them and produces a
# ResolutionImpossible once autoawq-kernels is added below. 0.2.9 has no
# torch constraint and only needs transformers>=4.45.0.
autoawq==0.2.9
# The autoawq PyPI package ships no compiled CUDA kernels itself: at runtime
# it tries to import the separately-packaged `awq_ext` extension and only
# falls back to JIT-compiling its dequant kernel via Triton if that import
# fails. The Triton fallback hit a CompilationError under the transformers
# regression described above; installing autoawq-kernels gives autoawq the
# compiled `awq_ext` module so it skips the Triton path entirely regardless.
# 0.0.9 is the last release (project archived) and needs torch>=2.5.1, hence
# the torch 2.6.0 base image.
autoawq-kernels==0.0.9
